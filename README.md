```

$casino = new Casino();

$user = User::getUser($trans['user_id']);

if($user->isBlocked()){
$trans_amount = $trans['amount'];

}
if($trans_amount > 0){
$result = $casino->changeUserBalance($user, $trans_amount);

if($casino->okResult($result)){
DB::delete($table, ['id' => $tid], $user->getId());

}
```
